					@Directive({ selector: '[clickOutside]' })
export class ClickOutsideDirective implements OnChanges, OnDestroy {
  @Input() isActive: boolean;
  // ...
  ngOnChanges(changes: SimpleChanges): void {
	const isActive = changes.isActive.currentValue;
	if (isActive) {
	  this.startClickListening();
	} else if (isActive !== undefined) {
	  this.stopClickListening();
	}
  }
  private startClickListening() {
    document.body.addEventListener('click', this.onClickBody);
  }
  private stopClickListening() {
    document.body.removeEventListener('click', this.onClickBody);
  }
  ngOnDestroy(): void {
    this.stopClickListening();
  }
  // ...
}
