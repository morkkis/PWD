  const response = await fetch('http://my-bank/data', request);
  return response.json();

Typescript

Same request TS way

						
async function getAccountData(request: AccountRequest): AccountResponse {
  const response = await fetch('http://my-bank/data', request);
  return response.json();
}

const request = new AccountRequest(12345);
const accountData = await getAccountData(request);

const firstTransaction = accountData.data.transactions[0];

<T> Generics

					
export class GroupOfPeople<T extends Person> {
  constructor(public group: T[]) {}
  makeThemTalk() {
    for (const person of this.group) {
      person.talk();
    }
  }
}

Spec file structure


describe('test unit', () => {
  beforeAll(() => {
    // once before the unit
  });
  beforeEach(() => {
    // once before every test
  });
  afterEach(() => {
    // once after every test
  });
  afterAll(() => {
    // once after the unit
  });
});

Given, When, Then

Explicit is better than implicit.
PEP20 - Python Zen Code

it('should calculate 115USD for 100USD money and 15% tip', () => {
  // given
  expect.hasAssertions();
  const calculator = new SimpleTipCalculator();

  // when
  const result = calculator.calc(100, 15);

  // then
  expect(result).toBe(115);
});

Before Each


describe('simple calculator', () => {
  let calculator: SimpleTipCalculator;
  beforeEach(() => {
    calculator = new SimpleTipCalculator();
  });
  
  it('should calculate 115USD for 100USD money and 15% Tip', () => {
    // given
    expect.hasAssertions();
    // when
    const result = calculator.calc(100, 15);
    // then
    expect(result).toBe(115);
  });
});
