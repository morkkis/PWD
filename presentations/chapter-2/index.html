					function Cat (name, breed, age, gender) {
  this.name = name;
  this.breed = breed;
  this.age = age;
  this.gender = gender;
  
  this.getName = function getName () {
    return this.name;
  };
  this.getAge = function getAge () {
    return this.age;
  };
};

console.log(cat.getName()); // Garfield
console.log(cat.getAge()); // 10
